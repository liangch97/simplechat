#include <iostream>
using namespace std;

int a[1005], n;

void heapify(int size, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    
    if (left < size && a[left] > a[largest]) largest = left;
    if (right < size && a[right] > a[largest]) largest = right;
    
    if (largest != i) {
        swap(a[i], a[largest]);
        heapify(size, largest);
    }
}

void heapSort() {
    // 建堆
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(n, i);
    }
    
    // 依次取出堆顶
    for (int i = n - 1; i > 0; i--) {
        swap(a[0], a[i]);
        heapify(i, 0);
    }
}

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    heapSort();
    
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << a[i];
    }
    cout << endl;
    return 0;
}
